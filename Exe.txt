import os
import requests
import shutil
import cloudpickle as cp
if not os.path.exists('random1.pkl'):
    print("Downloading model...")
    url = "https://drive.google.com/file/d/1KWaUhCUQvDaUrPEHzEB00uTZaWbAT3o9/view"
    r = requests.get(url, stream=True)
    with open('./random1.pkl', 'wb') as f:
        for chunk in r.iter_content(chunk_size=1024):
            if chunk:
                f.write(chunk)
    print("Finished downloading model.")

model = cp.load(open('C:\Users\dasai\Downloads\random.pkl','rb'))
